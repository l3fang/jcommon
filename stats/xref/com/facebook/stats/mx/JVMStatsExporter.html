35  /**
36   * Exports JVM stats so that they are available as counter values in fb303 stats.
37   * 
38   */
39  public class JVMStatsExporter {
40    private final Stats stats;
41    private final Pattern statNamePattern;
42    private static final String NAME_PREFIX = "jvm";
43    private static final Set<? extends OpenType> NUMERIC_TYPES = new ImmutableSet.Builder<SimpleType>()
44      .add(SimpleType.BYTE)
45      .add(SimpleType.SHORT)
46      .add(SimpleType.INTEGER)
47      .add(SimpleType.LONG)
48      .add(SimpleType.FLOAT)
49      .add(SimpleType.DOUBLE)
50      .build();

52    /**
53     * Creates an instance and exports the counters from the beans matching the supplied name
54     * pattern.
55     * 
56     *
57     * @param stats The stats instance to which all the dynamic counters will be registered.
58     * @param statNamePattern A regex {@link java.util.regex.Pattern} that is matched against all
59     * generated counter key names. Only the counters that match the supplied pattern are added to 
60     * Stats. Note that all stats have the prefix <code>"jvm."</code>.
61     * @param beanNamePatterns the object name patterns used to discover the Mbeans.
62     * 
63     * @throws JMException if the beanNamePattern is not a valid bean name pattern.
64     * @throws java.util.regex.PatternSyntaxException if the syntax of statNamePattern is invalid
65     * 
66     * @see ObjectName for details on the syntax of beanNamePattern 
67     */
68    public JVMStatsExporter(
69      Stats stats,
70      String statNamePattern,
71      String... beanNamePatterns) 
72        throws JMException {
73      this.stats = stats;
74      this.statNamePattern = Pattern.compile(statNamePattern);
75      for (String beanNamePattern : beanNamePatterns) {
76        exportNumericAttributes(new ObjectName(beanNamePattern));      
77      }
78    }

80    /**
81     * Creates an instance and exports all the counters from all the platform beans.
82     * 
83     * @param stats The stats instance to which all dynamic counters will be registered.
84     * @throws JMException unexpected, indicates a coding/error bug.
85     */
86    public JVMStatsExporter(Stats stats) throws JMException {
87      this(stats, ".*", "java.lang:type=*,*");
88    }

90    /**
91     * Exports all the numeric attributes of beans matching the supplied MXBean name pattern.
92     * Also, discovers the numeric properties of the attributes of type CompositeData and registers
93     * them as well.
94     * 
95     * @param beanNamePattern the bean name pattern used to discover the beans.
96     * @see javax.management.ObjectName for bean name pattern syntax
97     * @see java.lang.management for the list of java platform mbeans
98     * 
99     * @throws JMException if there are errors querying MBeans or information on them
100    */
101   public void exportNumericAttributes(ObjectName beanNamePattern) throws JMException {
102     MBeanServer beanServer = ManagementFactory.getPlatformMBeanServer();
103     Set<ObjectName> beanNames = beanServer.queryNames(beanNamePattern, null);
104     
105     // Iterate through all beans and register their numeric properties with Stats
106     for (ObjectName beanName : beanNames) {
107       MBeanInfo beanInfo = beanServer.getMBeanInfo(beanName);
108       
109       // Iterate through all bean attributes
110       for (MBeanAttributeInfo attributeInfo : beanInfo.getAttributes()) {
111         if (attributeInfo.isReadable()) {
112           // Figure out the open type for the attribute
113           OpenType<?> openType = null;
114           if (attributeInfo instanceof OpenMBeanAttributeInfo) {
115             openType = ((OpenMBeanAttributeInfo) attributeInfo).getOpenType();
116           } else {
117             // Sometimes the open mbean info is available in the descriptor
118             Object obj = attributeInfo.getDescriptor().getFieldValue("openType");
119             if (obj != null && obj instanceof OpenType) {
120               openType = (OpenType) obj;
121             }
122           }
123           // once the open type is found, figure out if it's a numeric or composite type
124           if (openType != null) {
125             if (NUMERIC_TYPES.contains(openType)) {
126               // numeric attribute types are registered with callbacks that simply 
127               // return their value 
128               addStatIfMatches(
129                 getStatName(beanName, attributeInfo.getName()),
130                 new MBeanLongAttributeFetcher(beanServer, beanName, attributeInfo.getName())
131               );
132             } else if (openType instanceof CompositeType) {
133               // for composite types, we figure out which properties of the composite type 
134               // are numeric and register callbacks to fetch those composite type attributes
135               CompositeType compositeType = (CompositeType) openType;
136               
137               for (String key : compositeType.keySet()) {
138                 if (NUMERIC_TYPES.contains(compositeType.getType(key))) {
139                   addStatIfMatches(
140                     getStatName(beanName, attributeInfo.getName(), key),
141                     new MBeanLongCompositeValueFetcher(
142                       beanServer, beanName,
143                       attributeInfo.getName(), key
144                     )
145                   );
146                 }
147               }
148             }
149           }
150         }  
151       }
152     }
153   }

155   /**
156    * Adds the supplied stat if it matches the statName pattern.
157    * 
158    * @param statName the name of the stat
159    * @param fetcher the fetcher that can fetch the value of the stat at run time.
160    */
161   private void addStatIfMatches(String statName, Callable<Long> fetcher) {
162     if (statNamePattern.matcher(statName).matches()) {
163       stats.addDynamicCounter(statName, fetcher);
164     }
165   }

167   /**
168    * Returns a stat name for the given set of parameters.
169    * 
170    * @param beanName the MBean name to which the stat belongs
171    * @param attributeNames the mbean attribute name followed by any nested attribute names for
172    * composite types
173    * 
174    * @return the stat name
175    */
176   private static String getStatName(ObjectName beanName, String... attributeNames) {
177     StringBuilder builder = new StringBuilder(NAME_PREFIX);
178     String value = beanName.getKeyProperty("type");
179     if (value != null) {
180       builder.append('.').append(value);
181     }
182     value = beanName.getKeyProperty("name");
183     if (value != null) {
184       builder.append('.').append(value);
185     }
186     for (String attributeName : attributeNames) {
187       builder.append('.').append(attributeName);
188     }
189     return builder.toString().replace(' ', '_');
190   }

192   /**
193    * A base class that fetches an mbean attribute value.
194    */
195   private static class MBeanAttributeFetcher {
196     private final MBeanServer mBeanServer;
197     private final ObjectName beanName;
198     private final String attribute;
199 
200     protected MBeanAttributeFetcher(MBeanServer mBeanServer, 
201                                     ObjectName beanName, 
202                                     String attribute) {
203       this.mBeanServer = mBeanServer;
204       this.beanName = beanName;
205       this.attribute = attribute;
206     }
207 
208     protected Object getAttributeValue() throws Exception {
209       return mBeanServer.getAttribute(beanName, attribute);
210     }
211   }

213   /**
214    * Fetches attribute value as long
215    */
216   private static class MBeanLongAttributeFetcher 
217     extends MBeanAttributeFetcher 
218     implements Callable<Long> {
219     private MBeanLongAttributeFetcher(
220         MBeanServer mBeanServer, ObjectName beanName, String attribute) {
221       super(mBeanServer, beanName, attribute);
222     }
223 
224     @Override
225     public Long call() throws Exception {
226       Object obj = getAttributeValue();
227       if(obj instanceof Number) {
228         return ((Number)obj).longValue();
229       }
230       return -1l;
231     }
232   }

234   /**
235    * Fetches an attribute of an mbean attribute that is of composite type.
236    */
237   private static class MBeanLongCompositeValueFetcher 
238       extends MBeanAttributeFetcher implements Callable<Long> {
239     private final String itemName;
240 
241     private MBeanLongCompositeValueFetcher(
242         MBeanServer mBeanServer, ObjectName beanName, String attribute, String itemName) {
243       super(mBeanServer, beanName, attribute);
244       this.itemName = itemName;
245     }
246 
247     @Override
248     public Long call() throws Exception {
249       Object obj = getAttributeValue();
250       if(obj instanceof CompositeData) {
251         obj = ((CompositeData)obj).get(itemName);
252         if (obj instanceof Number) {
253           return ((Number)obj).longValue();
254         }
255       }
256       return -1l;
257     }
258   }
